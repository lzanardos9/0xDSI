def rjson(keys=None):
    if keys is None:
        keys = ["key1", "key2"]
    return json.dumps({k: f"val_{random.randint(1,100)}" for k in keys})

def write_rows(table, rows):
    if rows:
        df = spark.createDataFrame(rows)
        df.write.mode("append").saveAsTable(table)
        print(f"  {table}: {len(rows)} rows")
